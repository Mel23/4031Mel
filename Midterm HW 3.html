def gaussianPDF(mu, sigma, x):
    return 1./(sigma * np.sqrt(2 * np.pi)) * np.exp( - (x - mu)**2 / (2 * sigma**2) )

def logLikelihoodGaussian(mu, sigma, cutoutdata):
    pxs = np.log(gaussianPDF(mu, sigma, cutoutdata))
    return np.array(pxs).sum()
